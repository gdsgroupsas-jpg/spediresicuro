# ============================================
# OAUTH PROVIDERS (Autenticazione Social)
# ============================================
# Configurazione per accesso tramite Google, GitHub, Facebook

# Google OAuth
# Ottieni le credenziali da: https://console.cloud.google.com/apis/credentials
# Callback URL: http://localhost:3000/api/auth/callback/google (sviluppo)
# Callback URL: https://yourdomain.com/api/auth/callback/google (produzione)

# Ottieni le credenziali da: https://github.com/settings/developers

# Authorization callback URL: https://yourdomain.com/api/auth/callback/github (produzione)

GITHUB_CLIENT_SECRET=your-github-client-secret

# Facebook OAuth
# Ottieni le credenziali da: https://developers.facebook.com/apps/
# Valid OAuth Redirect URIs: http://localhost:3000/api/auth/callback/facebook (sviluppo)
# Valid OAuth Redirect URIs: https://yourdomain.com/api/auth/callback/facebook (produzione)
